import streamlit as st
import yfinance as yf
import plotly.graph_objects as go
from datetime import datetime, timedelta

st.set_page_config(page_title="글로벌 시가총액 TOP 10 주가 변화", layout="wide")

st.title("📈 글로벌 시가총액 TOP 10 기업의 최근 1년간 주가 변화")

# 글로벌 시가총액 TOP 10 기업 (2025년 기준, 사우디 아람코 제외 - 데이터 문제로)
top10_tickers = {
    "Apple": "AAPL",
    "Microsoft": "MSFT",
    "NVIDIA": "NVDA",
    "Alphabet (Google)": "GOOGL",
    "Amazon": "AMZN",
    "Berkshire Hathaway": "BRK-B",
    "Meta (Facebook)": "META",
    "Eli Lilly": "LLY",
    "TSMC": "TSM",
    "Visa": "V"  # 10위권에 자주 들어가는 비자
}

# 날짜 범위: 오늘부터 1년 전까지
end_date = datetime.today()
start_date = end_date - timedelta(days=365)

fig = go.Figure()

for name, ticker in top10_tickers.items():
    try:
        data = yf.download(ticker, start=start_date, end=end_date)
        if data.empty:
            st.warning(f"{name}({ticker}) 데이터가 없습니다.")
            continue
        fig.add_trace(go.Scatter(
            x=data.index,
            y=data['Close'],
            mode='lines',
            name=name,
            hovertemplate='%{y:.2f} USD<br>%{x|%Y-%m-%d}'
        ))
    except Exception as e:
        st.error(f"{name}({ticker}) 데이터를 불러오는 중 오류 발생: {e}")

fig.update_layout(
    title="글로벌 시가총액 TOP 10 기업 주가 변화 (최근 1년)",
    xaxis_title="날짜",
    yaxis_title="종가 (USD)",
    template="plotly_white",
    hovermode="x unified",
    legend_title="기업명",
    height=600
)

st.plotly_chart(fig, use_container_width=True)
