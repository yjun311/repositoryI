import streamlit as st
import yfinance as yf
import plotly.graph_objects as go
from datetime import datetime, timedelta

# 글로벌 시가총액 TOP 10 기업 (2025 기준 추정)
top10_tickers = {
    "Apple": "AAPL",
    "Microsoft": "MSFT",
    "Saudi Aramco": "2222.SR",  # 사우디 아람코 (Tadawul)
    "NVIDIA": "NVDA",
    "Alphabet (Google)": "GOOGL",
    "Amazon": "AMZN",
    "Berkshire Hathaway": "BRK-B",
    "Meta (Facebook)": "META",
    "Eli Lilly": "LLY",
    "TSMC": "TSM"
}

st.title("📈 글로벌 시가총액 TOP 10 기업의 최근 1년간 주가 변화")

# 날짜 범위 설정
end_date = datetime.today()
start_date = end_date - timedelta(days=365)

# 데이터 다운로드 및 시각화
fig = go.Figure()

for name, ticker in top10_tickers.items():
    try:
        data = yf.download(ticker, start=start_date, end=end_date)
        fig.add_trace(go.Scatter(
            x=data.index,
            y=data['Close'],
            mode='lines',
            name=name
        ))
    except Exception as e:
        st.warning(f"{name} ({ticker}) 데이터를 가져오는 데 실패했습니다: {e}")

# 플롯 설정
fig.update_layout(
    title="글로벌 시가총액 TOP 10 주가 변화 (최근 1년)",
    xaxis_title="날짜",
    yaxis_title="종가 (USD)",
    template="plotly_dark",
    hovermode="x unified"
)

st.plotly_chart(fig, use_container_width=True)
